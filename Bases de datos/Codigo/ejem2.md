    public static int inversor(int n) {

        //Variables
        int mov = 0;

        while (n > 0) {
            int num = n % 10; // Saca el resto que es el numero a usar
            mov = mov * 10 + num; // Mueve a la izquierda el numero y le suma el resto
            n = n / 10; // Elimina el primer numero por la izquierda
        }
        return mov;
    } //fin metodo
